Utils API Reference

HandyGridPaginator (utils.paginator)
method desc
addParams(params: HandyGridPaginatorInterface) type: 

          {
            first?: number;
            previous?: number;           // Previous page number
            current?: number;            // Current page number
            next?: number;               // Next page number
            last?: number;               // Last page number
            pageCountAll?: number;       // Number of pages
            pageCountDisplay?: number;   // Number of pages max/display (buttons)
            itemCountPerPage?: number;   // Number of items per page
            itemCount?: number;          // Number of rows
          }
getItemCountPerPage() returns number of items per page
getFirst() returns page number
getPrevious() returns previous page number
getCurrent() returns current page number
getNext() returns next page number
getLast() returns last page number
getPageNumbers() returns page numbers
process() calculate propertis (pageCountAll, pageCountDisplay, last, next, previous)
HandyGridControlParams (utils.params)
method desc
addParams(params: HandyGridControlParamsInterface) type: 

          {
            current?: number;           // Current page number
            itemCountPerPage?: number;  // Number of items per page
            sortType?: number;          // sort type: 0, 1, 2
            sortField?: string;         // sort field name
          }
getParams() returns object<HandyGridControlParamsInterface>
getParamsAsQuery() returns string, e.g. ?current=2&itemCountPerPage=10&sortType=1&sortField=name
setPaginatorService(paginatorService: HandyGridPaginator) type: HandyGridPaginator
